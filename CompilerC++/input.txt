int a = 2;

void add(int n) {
    a = a + n;
}

void mult(int n) {
    a = a * n;
}

void sub(int n, int m) {
    a = a - (n + m);
}

void div(int n) {
    a = a / n;
}

void main() {
    short n = 2;
    short choice = 2;
    switch(choice - 1) {
        case 0:
            add(n);
            break;
        case 1:
            mult(-1);
        case 2:
            sub(1, - 1);
            break;
            a = 999;
        default:
            div(n);
            break;
    }
}
