void main() {
    // Присваивание с неявным приведением (должны быть предупреждения)
    short s1 = 50000;           // Предупреждение: обрезка int к short
    int i1 = 3000000000;        // Предупреждение: обрезка long к int
    
    // Корректные присваивания с приведением
    short s2 = 100;             // Без предупреждения (входит в диапазон)
    int i2 = 1000;              // Без предупреждения
    long l1 = 100000;           // Без предупреждения
    
    // Присваивание между переменными разных типов
    int source_int = 500;
    short target_short = source_int;    // Предупреждение: int к short
    
    long source_long = 1000000;
    int target_int = source_long;       // Предупреждение: long к int
    
    // Множественные присваивания
    short a, b, c;
    a = 1000;                   // Предупреждение: int к short
    b = a;                      // Без предупреждения (один тип)
    c = 50000;                  // Предупреждение: int к short
}
