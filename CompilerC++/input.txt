int a = 2;

void add(int n) {
    a = a + n;
}

void mult(int n) {
    a = a * n;
}

void main() {
    int n = 10;
    int choice = 1;

    switch (choice) {
        case 0:
            add(n);
            break;
        case 1:
            mult(n);
            break;
        default:
            a = 999;
            break;
    }
}
